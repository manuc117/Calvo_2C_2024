94static void mostrarDistanciaTask(void *pvParameter){
95 while (true){
96 uint16_t distancia;
97
98 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
99
100 if(ACTIVAR == true){
101
102 distancia = HcSr04ReadDistanceInCentimeters();
103 if(distancia < 10){
104 LedsOffAll();
105 }
106 else if(distancia > 10 && distancia < 20){
107 LedOn(LED_1);
108 LedOff(LED_2);
109 LedOff(LED_3);
110 }
111 else if(distancia > 20 && distancia < 30){
112 LedOn(LED_1);
113 LedOn(LED_2);
114 LedOff(LED_3);
115 }
116 else if(distancia > 30){
117 LedOn(LED_1);
118 LedOn(LED_2);
119 LedOn(LED_3);
120 }
121
122 if(HOLD == false){
123 LcdItsE0803Write(distancia);
124 }
125 }
126 else if(ACTIVAR == false){
127 LcdItsE0803Off();
128 LedsOffAll();
129 }
130 }
131}
