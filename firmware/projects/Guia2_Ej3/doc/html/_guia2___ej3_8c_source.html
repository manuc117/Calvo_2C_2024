97static void mostrarDistanciaTask(void *pvParameter){
98 while (true){
99
100 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
101
102 if(ACTIVAR == true){
103
104 distancia = HcSr04ReadDistanceInCentimeters();
105 if(distancia < 10){
106 LedsOffAll();
107 }
108 else if(distancia > 10 && distancia < 20){
109 LedOn(LED_1);
110 LedOff(LED_2);
111 LedOff(LED_3);
112 }
113 else if(distancia > 20 && distancia < 30){
114 LedOn(LED_1);
115 LedOn(LED_2);
116 LedOff(LED_3);
117 }
118 else if(distancia > 30){
119 LedOn(LED_1);
120 LedOn(LED_2);
121 LedOn(LED_3);
122 }
123
124 if(HOLD == false){
125 LcdItsE0803Write(distancia);
126 }
127
128 UartSendString(UART_PC, (const char*)UartItoa(distancia, 10));
129 UartSendString(UART_PC, " cm\r\n");
130 }
131 else if(ACTIVAR == false){
132 LcdItsE0803Off();
133 LedsOffAll();
134 }
135 }
136}
